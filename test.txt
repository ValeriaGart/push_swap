int ft_str_check(char **av, int ac)
{
    int i;
    int j;
    int n;

    i = 1;
    j = 0;
    n = 0;
    while(i < ac)
    {
        if (!av[i][j])
        {
            i++;
            j = 0;
        }
        else
        {
            while (av[i][j] == ' ')
                j++;
            if (ft_strchr("+-", av[i][j]))
            {
                if (!av[i][j + 1] || !ft_strchr("0123456789", av[i][j + 1]))
                    ft_error_exit("Wrong digits", 12);
                j++;
            }
            if (ft_strchr("0123456789", av[i][j]))
                n++;
            while (ft_strchr("0123456789", av[i][j]))
                j++;
            if (!ft_strchr("1234567890+- ", av[i][j]));
                ft_error_exit("Wrong digits", 12);
        }
    }
    return (n);
}

void    ft_check_str(int ac, char **av)
{
    int **num_arr;
    int am_nums;

    am_nums = ft_str_check(av, ac);
}
